Database Schema

-- Core User and Company Management
CREATE TABLE users (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    email VARCHAR(255) UNIQUE NOT NULL,
    company_id UUID NOT NULL,
    role VARCHAR(50) NOT NULL DEFAULT 'analyst'
);

CREATE TABLE companies (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name VARCHAR(255) NOT NULL,
    industry VARCHAR(100) NOT NULL,
    employee_count INTEGER,
    annual_revenue DECIMAL(15,2),
    sbti_sector VARCHAR(100)
);

-- ESPK Assessment Framework
CREATE TABLE espk_assessments (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    company_id UUID NOT NULL REFERENCES companies(id),
    environmental_score DECIMAL(5,2) CHECK (environmental_score BETWEEN 0 AND 100),
    social_score DECIMAL(5,2) CHECK (social_score BETWEEN 0 AND 100),
    physical_score DECIMAL(5,2) CHECK (physical_score BETWEEN 0 AND 100),
    knowledge_score DECIMAL(5,2) CHECK (knowledge_score BETWEEN 0 AND 100),
    overall_score DECIMAL(5,2) CHECK (overall_score BETWEEN 0 AND 100),
    assessment_data JSONB NOT NULL
);

-- Strategy and Implementation Intelligence
CREATE TABLE strategies (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name VARCHAR(255) NOT NULL,
    sector VARCHAR(100) NOT NULL,
    trl_level INTEGER CHECK (trl_level BETWEEN 1 AND 9),
    trl_category VARCHAR(20) CHECK (trl_category IN ('proven', 'emerging', 'experimental')),
    avg_implementation_cost DECIMAL(12,2),
    success_rate DECIMAL(3,2) CHECK (success_rate BETWEEN 0 AND 1)
);

Security Architecture

Authentication & Authorization

  • JWT Tokens

    HS256 signing with 24-hour expiration and refresh token rotation

  • Company-Scoped Access

    Users can only access their company's data with role-based permissions

  • API Rate Limiting

    100 requests/minute per user, 1000/minute per company

Data Protection

  • Encryption at Rest

    PostgreSQL transparent data encryption for sensitive company data

  • TLS 1.2+ Transit

    All API communications with certificate pinning

Input Validation

  • Pydantic + Zod

    Comprehensive validation across TypeScript/Python boundary

  • Range Validation

    ESPK scores strictly validated within 0-100 range

  • Data Anonymization

    Peer intelligence shows only aggregated statistics
